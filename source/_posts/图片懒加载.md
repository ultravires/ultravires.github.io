图片懒加载实现原理：当图片元素出现在视图中时加载图片资源。

```html
<!DOCTYPE html>
<html lang="zh">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>图片懒加载示例</title>
  <style>
    body {
      background-color: aliceblue;
    }
    .scrollContainer {
      width: 800px;
      height: 600px;
      background-color: white;
      overflow: auto;
    }
    .content {
      height: 1000px;
    }
  </style>
</head>
<body>
  <div class="scrollContainer">
    <div class="content">这里有很长的内容，请往下滚动</div>
    <img data-src="https://source.unsplash.com/random" alt="图片" />
  </div>
  <script type="module" src="./index.js"></script>
</body>
</html>
```

```js
const scrollContainer = document.querySelector('.scrollContainer');
const images = document.querySelectorAll('img');

scrollContainer.addEventListener('scroll', (event) => {
  Array.from(images).forEach((image) => {
    if (!image.dataset.loaded && isScrollIntoView(image)) {
      image.src = image.dataset.src;
      image.dataset.loaded = true;
    }
  });
});

/**
 * 判断元素是否滚动到视图内
 * @param {HTMLElement} el 
 */
function isScrollIntoView(el) {
  const rect = el.getBoundingClientRect();
  if (scrollContainer.offsetTop + scrollContainer.clientHeight > rect.top) {
    return true;
  }
  return false;
}
```
